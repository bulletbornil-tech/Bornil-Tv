function play() { const url = document.getElementById("url").value; const video = document.getElementById("video"); if (Hls.isSupported()) { const hls = new Hls(); hls.loadSource(url); hls.attachMedia(video); hls.on(Hls.Events.MANIFEST_PARSED, function () { video.play(); }); } else { video.src = url; video.play(); } }
